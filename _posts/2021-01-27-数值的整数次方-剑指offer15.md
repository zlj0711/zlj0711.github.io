最开始就是按部就班的分情况去做，但是我写的方法测试用例都能通过，但是超时

```c++
class Solution {
public:
    double myPow(double x, int n) {
        double res = 1;
        if(x == 1) return res;
        else if(x == -1){
            if(n % 2 == 0) return res;
            else return (-1) * res;
        }
        else{
            if(n == 0) return res;
            if(n > 0){
                while(n > 0){
                    res *= x;
                    n --;
                }
                return res;
            }
            if(n < 0){
                while(n < 0){
                    res /= x;
                    n ++;
                    if(res == 0) return 0;
                }
                return res;
            }
        }
        return res;        
    }
};
```
